    2. **`backend/app/services/today_service.py`** — aggregator:
       ```python
       """Today aggregator. Computes today's meals from active plan + completions + today's weight/workout.
       Source: TODAY-01..08."""
       from __future__ import annotations
       from datetime import datetime, date as date_t
       from zoneinfo import ZoneInfo
       from sqlalchemy import select
       from sqlalchemy.ext.asyncio import AsyncSession
       from app.models.plan import NutritionPlan
       from app.models.variant import WeeklyPlanVariant, Visibility
       from app.models.weight import WeightLog
       from app.models.workout import WorkoutLog
       from app.models.user import User
       from app.schemas.today import TodayResponse, MealEntry, MealMacro, TodayWeight, TodayWorkout

       MEAL_TYPES = ["breakfast", "lunch", "dinner", "snack"]

       def _greeting_period(now: datetime) -> str:
           h = now.hour
           if 5 <= h < 12: return "morning"
           if 12 <= h < 18: return "afternoon"
           if 18 <= h < 23: return "evening"
           return "night"

       async def build_today_payload(session: AsyncSession, user: User) -> TodayResponse:
           tz = ZoneInfo(user.timezone or "Europe/Rome")
           now = datetime.now(tz)
           today: date_t = now.date()
           # ISO weekday: Monday=0..Sunday=6 (we use isoweekday-1)
           day_of_week = today.weekday()

           # Find week_start (Monday of current week)
           from datetime import timedelta
           week_start = today - timedelta(days=day_of_week)

           # Active plan
           plan = (await session.scalars(
               select(NutritionPlan)
               .where(NutritionPlan.user_id == user.id, NutritionPlan.is_active.is_(True))
           )).first()

           meals: list[MealEntry] = []
           if plan:
               parsed = plan.parsed_json or {}
               # Phase 1 simple: emit meals from parsed sections, default variant 'default'
               # Plan 04 parser stores breakfast as MealOption | None, snacks as list[MealOption], lunches/dinners as dict
               breakfast = parsed.get("breakfast")
               if breakfast:
                   meals.append(MealEntry(
                       meal_type="breakfast", variant_key="default",
                       title=breakfast.get("title", "Colazione"),
                       macros=MealMacro(**(breakfast.get("macros") or {})),
                   ))
               # Lunches/dinners: pick first option from default key (Phase 2 weekly variant selector)
               for slot, target_dict_key in [("lunch", "lunches"), ("dinner", "dinners")]:
                   d = parsed.get(target_dict_key) or {}
                   options = d.get("default") or next(iter(d.values()), [])
                   if options:
                       opt = options[0]
                       meals.append(MealEntry(
                           meal_type=slot, variant_key=opt.get("key", "default"),
                           title=opt.get("title", slot.capitalize()),
                           macros=MealMacro(**(opt.get("macros") or {})),
                       ))
               # Snacks
               snacks = parsed.get("snacks") or []
               for sn in snacks:
                   meals.append(MealEntry(
                       meal_type="snack", variant_key=sn.get("key", "default"),
                       title=sn.get("title", "Spuntino"),
                       macros=MealMacro(**(sn.get("macros") or {})),
                   ))

               # Apply completion state from WeeklyPlanVariant
               variants = (await session.scalars(
                   select(WeeklyPlanVariant).where(
                       WeeklyPlanVariant.user_id == user.id,
                       WeeklyPlanVariant.week_start == week_start,
                       WeeklyPlanVariant.day_of_week == day_of_week,
                   )
               )).all()
               by_meal_type = {v.meal_type: v for v in variants}
               for m in meals:
                   v = by_meal_type.get(m.meal_type)
                   if v:
                       m.completed = v.completed

           # Today's weight
           w = (await session.scalars(
               select(WeightLog).where(WeightLog.user_id == user.id, WeightLog.date == today)
           )).first()
           weight_today = TodayWeight(id=str(w.id), weight_kg=float(w.weight_kg)) if w else None

           # Today's workout
           wk = (await session.scalars(
               select(WorkoutLog).where(WorkoutLog.user_id == user.id, WorkoutLog.date == today)
           )).first()
           workout_today = TodayWorkout(
               id=str(wk.id), trained=wk.trained, duration_min=wk.duration_min,
               calories_burned=wk.calories_burned, workout_type=wk.workout_type, notes=wk.notes,
           ) if wk else None

           return TodayResponse(
               date=today, day_of_week=day_of_week, greeting_period=_greeting_period(now),
               meals=meals, weight_today=weight_today, workout_today=workout_today,
           )

       async def complete_meal(
           session: AsyncSession, *, user: User, meal_type: str,
       ) -> WeeklyPlanVariant:
           if meal_type not in MEAL_TYPES:
               from app.core.exceptions import AppException
               raise AppException(400, "Tipo pasto non valido", "invalid_meal_type")
           tz = ZoneInfo(user.timezone or "Europe/Rome")
           now = datetime.now(tz)
           today = now.date()
           day_of_week = today.weekday()
           from datetime import timedelta
           week_start = today - timedelta(days=day_of_week)

           # Find active plan
           plan = (await session.scalars(
               select(NutritionPlan).where(
                   NutritionPlan.user_id == user.id, NutritionPlan.is_active.is_(True),
               )
           )).first()
           if not plan:
               from app.core.exceptions import AppException
               raise AppException(400, "Nessun piano attivo", "no_active_plan")

           # Find or create variant row
           variant = (await session.scalars(
               select(WeeklyPlanVariant).where(
                   WeeklyPlanVariant.user_id == user.id,
                   WeeklyPlanVariant.week_start == week_start,
                   WeeklyPlanVariant.day_of_week == day_of_week,
                   WeeklyPlanVariant.meal_type == meal_type,
               )
           )).first()
           if not variant:
               variant = WeeklyPlanVariant(
                   user_id=user.id, plan_id=plan.id,
                   week_start=week_start, day_of_week=day_of_week,
                   meal_type=meal_type, variant_key="default",
                   visibility=(Visibility.GROUP_SHARED if meal_type in ("lunch", "dinner") else Visibility.PRIVATE),
                   completed=True,
               )
               session.add(variant)
           else:
               variant.completed = True
               variant.version = variant.version + 1
           await session.commit()
           await session.refresh(variant)
           return variant
       ```

       ```python
       # weight_service.py
       from __future__ import annotations
       from datetime import date as date_t
       from decimal import Decimal
       from uuid import UUID
       from sqlalchemy import select
       from sqlalchemy.ext.asyncio import AsyncSession
       from sqlalchemy.exc import IntegrityError
       from app.core.exceptions import AppException
       from app.models.weight import WeightLog

       async def upsert_weight(session: AsyncSession, *, user_id: UUID, on_date: date_t, weight_kg: Decimal) -> WeightLog:
           existing = (await session.scalars(
               select(WeightLog).where(WeightLog.user_id == user_id, WeightLog.date == on_date)
           )).first()
           if existing:
               existing.weight_kg = weight_kg
               await session.commit()
               await session.refresh(existing)
               return existing
           w = WeightLog(user_id=user_id, date=on_date, weight_kg=weight_kg)
           session.add(w)
           try:
               await session.commit()
               await session.refresh(w)
               return w
           except IntegrityError:
               await session.rollback()
               # race retry
               return await upsert_weight(session, user_id=user_id, on_date=on_date, weight_kg=weight_kg)

       async def list_weights(session: AsyncSession, *, user_id: UUID) -> list[WeightLog]:
           rows = (await session.scalars(
               select(WeightLog).where(WeightLog.user_id == user_id).order_by(WeightLog.date.desc())
           )).all()
           return list(rows)

       async def update_weight(session: AsyncSession, *, user_id: UUID, weight_id: UUID, weight_kg: Decimal) -> WeightLog:
           w = (await session.scalars(
               select(WeightLog).where(WeightLog.id == weight_id, WeightLog.user_id == user_id)
           )).first()
           if not w:
               raise AppException(404, "Pesata non trovata", "not_found")
           w.weight_kg = weight_kg
           await session.commit()
           await session.refresh(w)
           return w

       async def delete_weight(session: AsyncSession, *, user_id: UUID, weight_id: UUID) -> None:
           w = (await session.scalars(
               select(WeightLog).where(WeightLog.id == weight_id, WeightLog.user_id == user_id)
           )).first()
           if not w:
               raise AppException(404, "Pesata non trovata", "not_found")
           await session.delete(w)
           await session.commit()
       ```

       ```typescript
       import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
       import { apiClient } from '@/services/api';
       import { db } from '@/db/dexie';

       export interface TodayMeal { meal_type: string; variant_key: string; title: string;
         macros: { kcal: number; protein_g: number; carbs_g: number; fat_g: number };
         completed: boolean }
       export interface TodayResponse {
         date: string; day_of_week: number;
         greeting_period: 'morning' | 'afternoon' | 'evening' | 'night';
         meals: TodayMeal[];
         weight_today: { id: string; weight_kg: number } | null;
         workout_today: { id: string; trained: boolean; duration_min: number | null;
                          calories_burned: number | null; workout_type: string | null;
                          notes: string | null } | null;
       }

       export function useToday() {
         return useQuery({
           queryKey: ['today'],
           queryFn: async () => {
             try {
               const resp = await apiClient.request<TodayResponse>({ url: '/api/today' });
               // Mirror to Dexie cache_today for offline
               await db.cache_today.put({
                 date: resp.date, user_id: '',
                 meals_completed: Object.fromEntries(resp.meals.map(m => [m.meal_type, m.completed])),
                 fetched_at: new Date().toISOString(),
               });
               return resp;
             } catch (e: any) {
               // Offline fallback
               const cached = await db.cache_today.toArray();
               if (cached.length > 0) {
                 return {
                   date: cached[0].date, day_of_week: 0, greeting_period: 'morning',
                   meals: [], weight_today: null, workout_today: null,
                 } as TodayResponse;
               }
               throw e;
             }
           },
         });
       }

       export function useCompleteMeal() {
         const qc = useQueryClient();
         return useMutation({
           mutationFn: async (meal_type: string) => {
             return apiClient.request({ url: `/api/today/meal/${meal_type}/complete`, method: 'POST' });
           },
           onMutate: async (meal_type) => {
             await qc.cancelQueries({ queryKey: ['today'] });
             const prev = qc.getQueryData<TodayResponse>(['today']);
             if (prev) {
               qc.setQueryData<TodayResponse>(['today'], {
                 ...prev,
                 meals: prev.meals.map(m => m.meal_type === meal_type ? { ...m, completed: true } : m),
               });
             }
             return { prev };
           },
           onError: (_e, _v, ctx) => {
             if (ctx?.prev) qc.setQueryData(['today'], ctx.prev);
           },
           onSettled: () => qc.invalidateQueries({ queryKey: ['today'] }),
         });
       }
       ```

       ```typescript
               await db.cache_weight_log.bulkPut(list.map(w => ({ ...w, user_id: '', updated_at: new Date().toISOString() })));

               const cached = await db.cache_weight_log.orderBy('date').reverse().toArray();

       export function useLogWeight() {
         const qc = useQueryClient();
         return useMutation({
           mutationFn: async (input: { date: string; weight_kg: number }) => {
             try {
               return await apiClient.request<WeightLog>({ url: '/api/weight', method: 'POST', data: input });
             } catch (e: any) {
               if (!navigator.onLine) {
                 await enqueueMutation({ endpoint: '/api/weight', method: 'POST', body: input });
                 return { id: 'pending-' + crypto.randomUUID(), date: input.date, weight_kg: input.weight_kg } as WeightLog;
               }
               throw e;
             }
           },
           onSuccess: () => {
             void qc.invalidateQueries({ queryKey: ['weights'] });
             void qc.invalidateQueries({ queryKey: ['today'] });
       ```
