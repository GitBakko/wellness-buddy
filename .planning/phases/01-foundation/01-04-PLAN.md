       ```python
       """Markdown text normalization pipeline. Source: PLAN-03, PITFALLS #6.

       Pipeline: utf-8-sig decode → CRLF→LF → NFC → NBSP→space → smart-punct→ASCII
       """
       from __future__ import annotations
       import re
       import unicodedata

       # NBSP variants (U+00A0, U+2007, U+202F, U+2060)
       _NBSP_REGEX = re.compile(r"[   ⁠]")

       _SMART_QUOTE_MAP = str.maketrans({
           "‘": "'", "’": "'", "‚": "'", "‛": "'",
           "“": '"', "”": '"', "„": '"', "‟": '"',
           "–": "-", "—": "-",  # en/em dash
           "…": "...",                # ellipsis
       })

       def normalize_md(raw_bytes: bytes) -> str:
           """Normalize MD bytes to clean parseable text.

           CRITICAL: 'utf-8-sig' decodes WITH BOM stripping; 'utf-8' does NOT.
           """
           text = raw_bytes.decode("utf-8-sig", errors="replace")
           text = text.replace("\r\n", "\n").replace("\r", "\n")
           text = unicodedata.normalize("NFC", text)
           text = _NBSP_REGEX.sub(" ", text)
           text = text.translate(_SMART_QUOTE_MAP)
           return text
       ```

       ```python
       """Tolerant MD plan parser. Stem-prefix matching, emoji-prefix tolerant.
       Source: PLAN-02, PLAN-04, PLAN-05, PLAN-06, RESEARCH Pattern 10."""
       from __future__ import annotations
       import re
       from dataclasses import dataclass, field
       from app.parsers.normalizer import normalize_md
       from app.parsers.plan_sections import parse_section
       from app.schemas.plan_parsed import PlanParsedSchema

       # Canonical stems (lowercase, accent-insensitive after NFC). Match by stem prefix.
       SECTION_STEMS: dict[str, str] = {
           "dati personali": "personal_data",
           "calcolo calorico": "macro_target",       # matches "CALCOLO CALORICO E MACRO TARGET"
           "struttura giornaliera": "daily_structure",
           "colazione": "breakfast",
           "pranzi": "lunches",
           "cene": "dinners",
           "spuntino pomeriggio": "snacks",
           "spuntino": "snacks",                       # tolerate "SPUNTINI" or "SPUNTINO"
           "supplementazione": "supplements",
           "proiezione peso": "weight_projection",
           "regole fondamentali": "rules",
       }

       _HEADING_RE = re.compile(r"^(#{1,6})\s*(.+?)\s*$", re.MULTILINE)
       _EMOJI_PREFIX_RE = re.compile(r"^[\W_]+", flags=re.UNICODE)

       def _heading_stem(heading_text: str) -> str:
           s = _EMOJI_PREFIX_RE.sub("", heading_text)
           s = re.sub(r"\s+", " ", s).strip().lower()
           return s

       def _split_sections(text: str) -> list[tuple[str, str]]:
           """Return [(heading, body), ...] preserving order."""
           parts: list[tuple[str, str]] = []
           matches = list(_HEADING_RE.finditer(text))
           for i, m in enumerate(matches):
               level = len(m.group(1))
               heading = m.group(2)
               # Phase 1: only level-2 (`##`) headings count as canonical sections;
               # nested `###` etc. are body content of their parent section.
               if level != 2:
                   continue
               start = m.end()
               # Find next level-2 heading
               end = len(text)
               for nxt in matches[i + 1:]:
                   if len(nxt.group(1)) == 2:
                       end = nxt.start()
                       break
               parts.append((heading, text[start:end].strip()))
           return parts

       @dataclass
       class ParseReport:
           warnings: list[str] = field(default_factory=list)
           unrecognized_headings: list[str] = field(default_factory=list)

       def parse_and_validate(raw_bytes: bytes) -> tuple[PlanParsedSchema, ParseReport]:
           """Parse + validate. Tolerant input → strict Pydantic v2 schema."""
           text = normalize_md(raw_bytes)
           sections = _split_sections(text)
           report = ParseReport()
           raw_dict: dict[str, object] = {}
           for heading, body in sections:
               stem = _heading_stem(heading)
               matched_target: str | None = None
               for canonical_stem, target in SECTION_STEMS.items():
                   if stem.startswith(canonical_stem):
                       matched_target = target
                       break
               if matched_target:
                   parsed_value, sub_warnings = parse_section(matched_target, body)
                   raw_dict[matched_target] = parsed_value
                   report.warnings.extend(sub_warnings)
               else:
                   report.unrecognized_headings.append(heading)
           # Pydantic v2 strict validation
           schema = PlanParsedSchema.model_validate(raw_dict)
           return schema, report
       ```

    4. **`backend/app/parsers/plan_sections.py`** — section-specific parsers (Phase 1 minimal: capture body as `{raw: "..."}` for most, but break out personal_data + macro_target with regex extraction so the schema isn't all-defaulted):
       ```python
       """Per-section parsers. Phase 1: minimal extraction for personal_data + macro_target;
       other sections store raw body for Phase 2 deeper parsing.

       Source: PLAN-02 sections list, RESEARCH Pattern 10 stub _parse_section."""
       from __future__ import annotations
       import re

       _NUMBER_RE = re.compile(r"(\d+(?:[.,]\d+)?)")

       def _to_float(s: str) -> float | None:
           m = _NUMBER_RE.search(s.replace(",", "."))
           return float(m.group(1)) if m else None

       def _to_int(s: str) -> int | None:
           f = _to_float(s)
           return int(f) if f is not None else None

       def _parse_personal_data(body: str) -> tuple[dict, list[str]]:
           data: dict = {}
           warnings: list[str] = []
           for line in body.splitlines():
               line = line.lstrip("- *").strip()
               low = line.lower()
               if low.startswith("nome"):
                   _, _, val = line.partition(":")
                   data["name"] = val.strip() or None
               elif low.startswith("età") or low.startswith("eta"):
                   _, _, val = line.partition(":")
                   data["age"] = _to_int(val)
               elif "peso attuale" in low:
                   _, _, val = line.partition(":")
                   data["current_weight_kg"] = _to_float(val)
               elif "peso obiettivo" in low or "target" in low:
                   _, _, val = line.partition(":")
                   data["target_weight_kg"] = _to_float(val)
           return data, warnings

       def _parse_macros(body: str) -> tuple[dict, list[str]]:
           data: dict = {}
           warnings: list[str] = []
           for line in body.splitlines():
               low = line.lower()
               if "kcal" in low and "kcal" not in data:
                   data["kcal"] = _to_int(line) or 0
               elif "protein" in low or "proteine" in low:
                   data["protein_g"] = _to_float(line) or 0
               elif "carbo" in low or "carboidrat" in low:
                   data["carbs_g"] = _to_float(line) or 0
               elif "grass" in low or "lipid" in low:
                   data["fat_g"] = _to_float(line) or 0
           return data, warnings

       def _parse_meal_options(body: str) -> tuple[dict[str, list[dict]], list[str]]:
           """Parse `### Opzione A` etc. Phase 1: weekday-keyed if `### Lunedì` style appears,
           else single key 'default' with list of options.
           """
           # Minimal Phase 1: capture as raw body wrapped in a placeholder key
           return {"default": [{"key": "raw", "title": body[:80] or "(vuoto)", "ingredients": [], "macros": {}}]}, []

       def _parse_breakfast(body: str) -> tuple[dict | None, list[str]]:
           if not body.strip():
               return None, []
           return {"key": "default", "title": body.split("\n", 1)[0].strip()[:200] or "Colazione",
                   "ingredients": [], "macros": {}}, []

       def _parse_snacks(body: str) -> tuple[list[dict], list[str]]:
           return [{"key": "default", "title": body.split("\n", 1)[0].strip()[:200] or "Spuntino",
                    "ingredients": [], "macros": {}}], []

       def _parse_raw_lines(body: str) -> tuple[list[dict] | list[str], list[str]]:
           lines = [ln.lstrip("- *").strip() for ln in body.splitlines() if ln.strip()]
           return lines, []

       def parse_section(target: str, body: str) -> tuple[object, list[str]]:
           if target == "personal_data":
               return _parse_personal_data(body)
           if target == "macro_target":
               return _parse_macros(body)
           if target == "breakfast":
               return _parse_breakfast(body)
           if target == "lunches" or target == "dinners":
               return _parse_meal_options(body)
           if target == "snacks":
               return _parse_snacks(body)
           if target == "supplements":
               result, warnings = _parse_raw_lines(body)
               return [{"raw": line} for line in result], warnings  # type: ignore[union-attr]
           if target == "weight_projection":
               result, warnings = _parse_raw_lines(body)
               return [{"raw": line} for line in result], warnings  # type: ignore[union-attr]
           if target == "rules":
               return _parse_raw_lines(body)
           if target == "daily_structure":
               result, warnings = _parse_raw_lines(body)
               return [{"raw": line} for line in result], warnings  # type: ignore[union-attr]
           # Fallback
           return {"raw": body[:500]}, ["unknown_target_" + target]
       ```

       ```python
       """Strict Pydantic v2 schema downstream of tolerant parser. Source: PLAN-06, RESEARCH Pattern 10."""
       from __future__ import annotations
       from pydantic import BaseModel, ConfigDict, Field

       class PersonalData(BaseModel):
           model_config = ConfigDict(extra="forbid")
           name: str | None = None
           age: int | None = None
           current_weight_kg: float | None = None
           target_weight_kg: float | None = None

       class Macros(BaseModel):
           model_config = ConfigDict(extra="forbid")
           kcal: int = 0
           protein_g: float = 0
           carbs_g: float = 0
           fat_g: float = 0

       class Ingredient(BaseModel):
           model_config = ConfigDict(extra="forbid")
           name: str
           quantity: float | None = None
           unit: str | None = None
           category: str | None = None

       class MealOption(BaseModel):
           model_config = ConfigDict(extra="forbid")
           key: str
           title: str
           ingredients: list[Ingredient] = Field(default_factory=list)
           macros: Macros = Field(default_factory=Macros)

       class PlanParsedSchema(BaseModel):
           model_config = ConfigDict(extra="forbid")
           personal_data: PersonalData | None = None
           macro_target: Macros = Field(default_factory=Macros)
           daily_structure: list[dict] = Field(default_factory=list)
           breakfast: MealOption | None = None
           lunches: dict[str, list[MealOption]] = Field(default_factory=dict)
           dinners: dict[str, list[MealOption]] = Field(default_factory=dict)
           snacks: list[MealOption] = Field(default_factory=list)
           supplements: list[dict] = Field(default_factory=list)
           weight_projection: list[dict] = Field(default_factory=list)
           rules: list[str] = Field(default_factory=list)
       ```

       ```python
       """Plan service: upload, activate, diff, assign. Source: PLAN-07..PLAN-10, RESEARCH Pattern 10."""
       from __future__ import annotations
       from datetime import datetime, timezone
       from uuid import UUID
       from sqlalchemy import select, update
       from sqlalchemy.ext.asyncio import AsyncSession
       from app.core.exceptions import AppException
       from app.models.plan import NutritionPlan
       from app.parsers.plan_parser import parse_and_validate, ParseReport
       from app.schemas.plan_parsed import PlanParsedSchema
       from app.services.audit_service import write_audit

       MAX_FILE_BYTES = 1 * 1024 * 1024  # 1 MB cap (V12)

       async def upload_plan(
           session: AsyncSession, *, user_id: UUID, name: str, raw_bytes: bytes,
       ) -> tuple[NutritionPlan, ParseReport]:
           if len(raw_bytes) > MAX_FILE_BYTES:
               raise AppException(400, "Il file supera il limite di 1 MB.", "too_large")
           try:
               schema, report = parse_and_validate(raw_bytes)
           except Exception as e:
               raise AppException(400, "Non sono riuscito a leggere il piano. Verifica che il formato sia corretto.", "parse_failed") from e
           plan = NutritionPlan(
               user_id=user_id,
               name=name,
               raw_md=raw_bytes.decode("utf-8-sig", errors="replace"),
               parsed_json=schema.model_dump(),
               is_active=False,
           )
           session.add(plan)
           await session.flush()
           await write_audit(session, actor_id=user_id, action="plan_upload",
                             target_type="nutrition_plan", target_id=plan.id, payload={"name": name})
           await session.commit()
           await session.refresh(plan)
           return plan, report

       async def list_plans(session: AsyncSession, *, user_id: UUID) -> list[NutritionPlan]:
           rows = (await session.scalars(
               select(NutritionPlan)
               .where(NutritionPlan.user_id == user_id)
               .order_by(NutritionPlan.uploaded_at.desc())
           )).all()
           return list(rows)

       async def activate_plan(session: AsyncSession, *, user_id: UUID, plan_id: UUID) -> NutritionPlan:
           plan = (await session.scalars(
               select(NutritionPlan).where(NutritionPlan.id == plan_id, NutritionPlan.user_id == user_id)
           )).first()
           if not plan:
               raise AppException(404, "Piano non trovato.", "not_found")
           # Atomically deactivate previous + activate this (avoid partial unique index conflict)
           await session.execute(
               update(NutritionPlan)
               .where(NutritionPlan.user_id == user_id, NutritionPlan.is_active.is_(True))
               .values(is_active=False)
           )
           plan.is_active = True
           await write_audit(session, actor_id=user_id, action="plan_activate",
                             target_type="nutrition_plan", target_id=plan.id, payload={})
           await session.commit()
           await session.refresh(plan)
           return plan

       async def diff_against_active(
           session: AsyncSession, *, user_id: UUID, candidate_plan_id: UUID,
       ) -> dict[str, list[str]]:
           candidate = (await session.scalars(
               select(NutritionPlan).where(NutritionPlan.id == candidate_plan_id, NutritionPlan.user_id == user_id)
           )).first()
           if not candidate:
               raise AppException(404, "Piano non trovato.", "not_found")
           active = (await session.scalars(
               select(NutritionPlan).where(NutritionPlan.user_id == user_id, NutritionPlan.is_active.is_(True))
           )).first()
           if not active:
               # No active plan → all candidate sections are added
               present = [k for k, v in (candidate.parsed_json or {}).items() if v]
               return {"added": present, "removed": [], "changed": []}
           added: list[str] = []
           removed: list[str] = []
           changed: list[str] = []
           a = active.parsed_json or {}
           c = candidate.parsed_json or {}
           for key in set(a.keys()) | set(c.keys()):
               in_a = key in a and a[key] not in (None, [], {}, "")
               in_c = key in c and c[key] not in (None, [], {}, "")
               if in_c and not in_a:
                   added.append(key)
               elif in_a and not in_c:
                   removed.append(key)
               elif in_a and in_c and a[key] != c[key]:
                   changed.append(key)
           return {"added": sorted(added), "removed": sorted(removed), "changed": sorted(changed)}

       async def admin_assign_plan(
           session: AsyncSession, *, admin_id: UUID, target_user_id: UUID, plan_id: UUID,
       ) -> NutritionPlan:
           plan = (await session.scalars(select(NutritionPlan).where(NutritionPlan.id == plan_id))).first()
           if not plan:
               raise AppException(404, "Piano non trovato.", "not_found")
           plan.user_id = target_user_id
           await write_audit(session, actor_id=admin_id, action="plan_assign",
                             target_type="nutrition_plan", target_id=plan.id,
                             payload={"target_user_id": str(target_user_id)})
           await session.commit()
           await session.refresh(plan)
           return plan
       ```

       ```typescript
         const access = useAuthStore.getState().accessToken;
         const fd = new FormData();
         fd.append('file', file);
         fd.append('name', name);
         const res = await fetch('/api/plans/upload', {
           method: 'POST', body: fd, credentials: 'include',
           headers: access ? { Authorization: `Bearer ${access}` } : {},
         });
         if (!res.ok) {
           const data = await res.json().catch(() => ({ detail: 'Errore', code: 'unknown' }));
           throw { response: { status: res.status, data } };
         }
         return res.json() as Promise<PlanUploadResponse>;
       }

       export async function activatePlan(planId: string): Promise<{ id: string; name: string; is_active: boolean }> {
         return apiClient.request({ url: `/api/plans/${planId}/activate`, method: 'POST' });
       }

       export async function diffPlan(planId: string): Promise<PlanDiffResponse> {
         return apiClient.request<PlanDiffResponse>({ url: `/api/plans/${planId}/diff` });
       }
       ```
