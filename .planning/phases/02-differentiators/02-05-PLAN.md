    ```python
        if unit == "qb":
    ```

    ```typescript
    /**
     * D-25 — multi-tab shopping list sync.
     *
     * BroadcastChannel where supported; window.focus listener fallback for iOS Safari Private mode.
     */
    type Listener<T> = (msg: T) => void;

    export function createSyncChannel<T>(name: string, listener: Listener<T>): () => void {
      if (typeof window !== 'undefined' && 'BroadcastChannel' in window) {
        try {
          const bc = new BroadcastChannel(name);
          const handler = (e: MessageEvent<T>) => listener(e.data);
          bc.addEventListener('message', handler);
          return () => {
            bc.removeEventListener('message', handler);
            bc.close();
          };
        } catch {
          // SecurityError in Safari Private mode — fall through to focus fallback
        }
      }
      const onFocus = () => listener({ type: 'focus_refetch' } as unknown as T);
      window.addEventListener('focus', onFocus);
      return () => window.removeEventListener('focus', onFocus);
    }

    export function postSyncMessage<T>(name: string, msg: T): void {
      if (typeof window !== 'undefined' && 'BroadcastChannel' in window) {
        try {
          const bc = new BroadcastChannel(name);
          try { bc.postMessage(msg); }
          finally { bc.close(); }
        } catch {
          // Private mode silent no-op — focus event in OTHER tab triggers refetch
        }
      }
    }
    ```

    ```typescript
    export function useToggleItem(weekStart: string) {
      const userId = useAuthStore.getState().user?.id ?? '';
      const qc = useQueryClient();
      return useMutation({
        mutationFn: async (input: { canonical_name: string; unit: string | null; checked: boolean }) => {
          try {
            return await apiClient.request<ShoppingResponse>({
              url: `/api/shopping/${weekStart}/check`,
              method: 'PATCH',
              data: input,

                endpoint: `/api/shopping/${weekStart}/check`,

                body: input,

              return null; // optimistic UI keeps the change locally

            throw e;
          }

        onMutate: async (input) => {
          await qc.cancelQueries({ queryKey: shoppingQueryKey(userId, weekStart) });
          const prev = qc.getQueryData<ShoppingResponse>(shoppingQueryKey(userId, weekStart));
          if (prev) {

              categories: prev.categories.map((c) => ({

          }
          return { prev };

      return useMutation({
        mutationFn: async () =>
          apiClient.request({
            url: `/api/shopping/${weekStart}/reset`,

        onError: () => {

        },

    export function composeTextExport(payload: ShoppingResponse): string {
      const lines: string[] = [];
      lines.push(`Lista spesa — settimana del ${payload.week_start}`);
      lines.push('');
      for (const cat of payload.categories) {
        if (cat.items.length === 0) continue;
        lines.push(`## ${cat.name}`);
        for (const it of cat.items) {
          const check = it.checked ? '✓' : '☐';
          const qty = it.quantity_it ? ` — ${it.quantity_it}` : '';
          lines.push(`${check} ${it.name_display}${qty}`);
        }
        lines.push('');
      }
      return lines.join('\n');
    }
    ```
